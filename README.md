```bash
npm install --save rrule-generator-react
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'rrule-generator-react'
import 'rrule-generator-react/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
